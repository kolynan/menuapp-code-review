```javascript
// Counts unique guests across a list of orders.
```

**Change 1 — `components/sessionHelpers.js` (NEW behaviour for closeSession):**

**Template field-by-field:**

```javascript
export async function closeSession(sessionId) {
  // UNCHANGED — existing TableSession close
  await base44.entities.TableSession.update(sessionId, {
    status: "closed",
    closed_at: new Date().toISOString()
  });

  // NEW (S267) — Bulk-close all orders belonging to this session so SOM
  // can filter them out by o.status === 'closed' instead of relying on
  // TableSession entity being loaded in the page.
  // Callers (verified 2026-04-14):
  //   - pages/StaffOrdersMobile/*.jsx
  //   - pages/PartnerTables/*.js (admin closes table — same semantics)
```

**Change 3 — `filteredGroups` (line 3792-3801) + `tabCounts` (line 3804-3818) — EXACT CODE:**

After Change 2, `served` orders stay in `visibleOrders`. But `hasActiveOrder` logic in `filteredGroups` marks group as inactive (because `isFinishStage` is true for `served`). We need a third predicate to keep group in Active tab until closeSession.

**filteredGroups — REPLACE lines 3792-3801 with:**
```javascript
const filteredGroups = orderGroups.filter(group => {
  const hasActiveOrder = group.orders.some(o => {
    const config = getStatusConfig(o);
    return !config.isFinishStage && o.status !== 'cancelled';
  });
  const hasActiveRequest = (group.requests || []).some(r => r.status !== 'resolved' && r.status !== 'cancelled');
  // NEW (S267): served-but-not-closed — keep in Active tab until closeSession
  const hasServedButNotClosed = group.orders.some(o => {
    const config = getStatusConfig(o);
    return config.isFinishStage && o.status !== 'closed' && o.status !== 'cancelled';
  });
  return activeTab === 'active'
    ? (hasActiveOrder || hasActiveRequest || hasServedButNotClosed)
    : (!hasActiveOrder && !hasActiveRequest && !hasServedButNotClosed);
});
```

**tabCounts — REPLACE lines 3804-3818 with:**
```javascript
const tabCounts = orderGroups.reduce((acc, group) => {
  const hasActiveOrder = group.orders.some(o => {
    const config = getStatusConfig(o);
    return !config.isFinishStage && o.status !== 'cancelled';
  });
  const hasActiveRequest = (group.requests || []).some(r => r.status !== 'resolved' && r.status !== 'cancelled');
  const hasServedButNotClosed = group.orders.some(o => {
    const config = getStatusConfig(o);
    return config.isFinishStage && o.status !== 'closed' && o.status !== 'cancelled';
  });
  if (hasActiveOrder || hasActiveRequest || hasServedButNotClosed) {
    acc.active += 1;
  } else {
    acc.completed += 1;
  }
  return acc;
}, { active: 0, completed: 0 });
```
